  function Human (age) {
    this.age = age
  }

  let greg = new Human(22)
  let thomas = new Human(24)

  console.log(greg) // this.age = 22
  console.log(thomas) // this.age = 24

    In the above code snippet, the object "Human" is defined, and the this keyword is used. this refers to the object. Inside the "Human" constructor code block, instead of writing this.age = age, you can write greg.age = age (where greg refers to the greg object). However, whenever you create a new object (thomas), it will set greg's age to whatever was passed to the constructor. Hence, this is why we need to use this.
